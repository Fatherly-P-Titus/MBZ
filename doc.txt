// Route for support page
app.get('/support', (req, res) => {
    res.sendFile(path.join(__dirname, '../frontend/support.html'));
});

// Route for legal page
app.get('/legal', (req, res) => {
    res.sendFile(path.join(__dirname, '../frontend/legal.html'));
});

    const products = [
